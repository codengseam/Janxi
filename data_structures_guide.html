class DynamicArray:
    """动态数组：自动扩容，摊还 O(1) append"""
    def __init__(self):
        self._capacity = 1
        self._size = 0
        self._data = [None] * self._capacity

    def __len__(self):          return self._size
    def __getitem__(self, i):
        if not 0 <= i < self._size: raise IndexError("越界")
        return self._data[i]
    def __setitem__(self, i, v):
        if not 0 <= i < self._size: raise IndexError("越界")
        self._data[i] = v

    def append(self, val):
        if self._size == self._capacity:
            self._resize(2 * self._capacity)   # 容量翻倍
        self._data[self._size] = val
        self._size += 1

    def insert(self, idx, val):
        """O(n)：idx 之后的元素全部右移一位"""
        if self._size == self._capacity:
            self._resize(2 * self._capacity)
        for i in range(self._size, idx, -1):
            self._data[i] = self._data[i - 1]
        self._data[idx] = val
        self._size += 1

    def delete(self, idx):
        """O(n)：idx 之后的元素左移一位"""
        val = self._data[idx]
        for i in range(idx, self._size - 1):
            self._data[i] = self._data[i + 1]
        self._size -= 1
        # 缩容：空间利用率不足 25% 时减半
        if self._size < self._capacity // 4 and self._capacity > 1:
            self._resize(self._capacity // 2)
        return val

    def _resize(self, new_cap):
        new_data = [None] * new_cap
        for i in range(self._size):
            new_data[i] = self._data[i]
        self._data = new_data
        self._capacity = new_cap

class PrefixSum:
    def __init__(self, nums):
        n = len(nums)
        self.pre = [0] * (n + 1)           # pre[0] = 0（哨兵）
        for i in range(n):
            self.pre[i + 1] = self.pre[i] + nums[i]

    def range_sum(self, l, r):             # [l, r] 区间和，O(1)
        return self.pre[r + 1] - self.pre[l]

def two_sum_sorted(arr, target):
    """有序数组双指针，O(n)"""
    lo, hi = 0, len(arr) - 1
    while lo < hi:
        s = arr[lo] + arr[hi]
        if s == target:   return (lo, hi)
        elif s < target:  lo += 1
        else:             hi -= 1
    return None

def max_subarray_sum(nums, k):
    """长度为 k 的最大子数组和，O(n)"""
    window = sum(nums[:k])
    best = window
    for i in range(k, len(nums)):
        window += nums[i] - nums[i - k]
        best = max(best, window)
    return best

# ─── 单链表 ───
class SinglyLinkedList:
    def __init__(self):
        self.dummy = ListNode(0)   # 头哨兵，简化边界处理
        self.size = 0

    def append(self, val):         # 尾插 O(n)（若无 tail 指针）
        cur = self.dummy
        while cur.next: cur = cur.next
        cur.next = ListNode(val)
        self.size += 1

    def prepend(self, val):        # 头插 O(1)
        node = ListNode(val, self.dummy.next)
        self.dummy.next = node
        self.size += 1

    def delete(self, val):         # 按值删除 O(n)
        cur = self.dummy
        while cur.next:
            if cur.next.val == val:
                cur.next = cur.next.next
                self.size -= 1
                return True
            cur = cur.next
        return False

    def reverse(self):             # 原地反转 O(n)
        prev, cur = None, self.dummy.next
        while cur:
            nxt = cur.next
            cur.next = prev
            prev = cur
            cur = nxt
        self.dummy.next = prev

    def has_cycle(self):           # Floyd 判环
        slow = fast = self.dummy.next
        while fast and fast.next:
            slow = slow.next
            fast = fast.next.next
            if slow is fast: return True
        return False

class DoublyLinkedList:
    def __init__(self):
        self.head = DNode()          # 头哨兵（最久未使用端）
        self.tail = DNode()          # 尾哨兵（最近使用端）
        self.head.next = self.tail
        self.tail.prev = self.head
        self.size = 0

    def add_to_tail(self, node):     # 插入到尾部 O(1)
        node.prev = self.tail.prev
        node.next = self.tail
        self.tail.prev.next = node
        self.tail.prev = node
        self.size += 1

    def remove(self, node):          # 移除任意节点 O(1)
        node.prev.next = node.next
        node.next.prev = node.prev
        self.size -= 1

    def pop_head(self):              # 移除头部第一个真实节点（最久）O(1)
        if self.head.next is self.tail: return None
        node = self.head.next
        self.remove(node)
        return node

    def move_to_tail(self, node):    # 移到尾部（标记为最近使用）O(1)
        self.remove(node)
        self.add_to_tail(node)

def merge_sorted(l1, l2):
    dummy = ListNode(0)
    cur = dummy
    while l1 and l2:
        if l1.val <= l2.val: cur.next, l1 = l1, l1.next
        else:                cur.next, l2 = l2, l2.next
        cur = cur.next
    cur.next = l1 or l2
    return dummy.next

    def __init__(self):
        self._data = []

        if self.is_empty(): raise IndexError("栈为空")

def is_valid_brackets(s: str) -> bool:
    stack = []
    mapping = {')': '(', ']': '[', '}': '{'}
    for ch in s:
        if ch in '([{':
            stack.append(ch)
        elif not stack or stack[-1] != mapping[ch]:
            return False
        else:
            stack.pop()
    return not stack

def next_greater_element(nums):
    n = len(nums)
    result = [-1] * n
    stack = []                    # 存储下标
    for i in range(n):
        # 当前元素比栈顶大 → 栈顶找到了"下一个更大元素"
        while stack and nums[stack[-1]] < nums[i]:
            result[stack.pop()] = nums[i]
        stack.append(i)
    return result

def largest_rectangle(heights):
    stack = [-1]                  # 哨兵
    heights.append(0)
    max_area = 0
    for i, h in enumerate(heights):
        while stack[-1] != -1 and heights[stack[-1]] >= h:
            height = heights[stack.pop()]
            width = i - stack[-1] - 1
            max_area = max(max_area, height * width)
        stack.append(i)
    heights.pop()
    return max_area

class MyQueue:
    def __init__(self):
        self.in_stack = []        # 入队专用
        self.out_stack = []       # 出队专用

    def push(self, x):
        self.in_stack.append(x)

    def pop(self):
        self._migrate()
        return self.out_stack.pop()

    def peek(self):
        self._migrate()
        return self.out_stack[-1]

    def _migrate(self):
        if not self.out_stack:    # 懒迁移：out 空时才将 in 全部倒入
            while self.in_stack:
                self.out_stack.append(self.in_stack.pop())
单调栈核心思想：维护一个单调递增/递减的栈。当新元素"破坏"单调性时，弹出栈顶并记录关系。所有元素最多入栈/出栈各一次，总时间复杂度 O(n)。

class CircularQueue:
    def __init__(self, k):
        self.buf = [None] * k
        self.head = self.tail = 0
        self.size = 0
        self.k = k

    def enqueue(self, val):
        if self.size == self.k: raise OverflowError("队列满")
        self.buf[self.tail] = val
        self.tail = (self.tail + 1) % self.k
        self.size += 1

    def dequeue(self):
        if self.size == 0: raise IndexError("队列空")
        val = self.buf[self.head]
        self.head = (self.head + 1) % self.k
        self.size -= 1
        return val

# ─── 单调队列——滑动窗口最大值 O(n) ───
def sliding_window_max(nums, k):
    dq = deque()                # 存储下标，保持单调递减
    result = []

    for i, num in enumerate(nums):
        # 移除超出窗口的元素
        while dq and dq[0] < i - k + 1:
            dq.popleft()
        # 维护单调性：移除比当前值小的元素
        while dq and nums[dq[-1]] < num:
            dq.pop()
        dq.append(i)
        # 窗口形成后记录最大值
        if i >= k - 1:
            result.append(nums[dq[0]])
    return result

# ─── BFS 模板（最短路径）───
def bfs_shortest_path(grid, start, end):
    """网格最短路径，grid[i][j]=0 可走，1 障碍"""
    rows, cols = len(grid), len(grid[0])
    queue = deque([(start, 0)])   # (坐标, 步数)
    visited = {start}
    dirs = [(0,1),(0,-1),(1,0),(-1,0)]

    while queue:
        (r, c), dist = queue.popleft()
        if (r, c) == end: return dist
        for dr, dc in dirs:
            nr, nc = r + dr, c + dc
            if (0 <= nr < rows and 0 <= nc < cols
                    and (nr, nc) not in visited
                    and grid[nr][nc] == 0):
                visited.add((nr, nc))
                queue.append(((nr, nc), dist + 1))
    return -1

# ─── 拉链法哈希表 ───
class HashMap:
    LOAD_FACTOR = 0.75

    def __init__(self, capacity=16):
        self.capacity = capacity
        self.size = 0
        self.buckets = [[] for _ in range(capacity)]   # 每个桶是链表（list of (k,v)）

    def _hash(self, key):
        return hash(key) % self.capacity

    def put(self, key, val):
        idx = self._hash(key)
        for i, (k, v) in enumerate(self.buckets[idx]):
            if k == key:
                self.buckets[idx][i] = (key, val)     # 更新
                return
        self.buckets[idx].append((key, val))           # 新增
        self.size += 1
        if self.size / self.capacity > self.LOAD_FACTOR:
            self._resize()

    def get(self, key):
        for k, v in self.buckets[self._hash(key)]:
            if k == key: return v
        raise KeyError(key)

    def remove(self, key):
        idx = self._hash(key)
        bucket = self.buckets[idx]
        for i, (k, v) in enumerate(bucket):
            if k == key:
                bucket.pop(i)
                self.size -= 1
                return
        raise KeyError(key)

    def _resize(self):
        old = self.buckets
        self.capacity *= 2
        self.buckets = [[] for _ in range(self.capacity)]
        self.size = 0
        for bucket in old:
            for k, v in bucket:
                self.put(k, v)            # 重新哈希

# ─── 开放寻址法（线性探测）───
class OpenHashMap:
    DELETED = object()              # 墓碑标记（防止搜索链断裂）

    def __init__(self, capacity=16):
        self.capacity = capacity
        self.size = 0
        self.keys = [None] * capacity
        self.vals = [None] * capacity

    def _probe(self, key):
        """线性探测，找到 key 或空槽"""
        h = hash(key) % self.capacity
        while self.keys[h] is not None:
            if self.keys[h] is not self.DELETED and self.keys[h] == key:
                return h
            h = (h + 1) % self.capacity
        return h

    def put(self, key, val):
        if self.size / self.capacity > 0.5: self._resize()
        h = hash(key) % self.capacity
        while self.keys[h] is not None and self.keys[h] is not self.DELETED:
            if self.keys[h] == key:
                self.vals[h] = val; return
            h = (h + 1) % self.capacity
        self.keys[h] = key; self.vals[h] = val; self.size += 1

    def get(self, key):
        h = self._probe(key)
        if self.keys[h] == key: return self.vals[h]
        raise KeyError(key)

    def remove(self, key):
        h = self._probe(key)
        if self.keys[h] == key:
            self.keys[h] = self.DELETED   # 墓碑，不能直接置 None
            self.size -= 1

    def _resize(self):
        old_keys, old_vals = self.keys, self.vals
        self.capacity *= 2
        self.keys = [None] * self.capacity
        self.vals = [None] * self.capacity
        self.size = 0
        for k, v in zip(old_keys, old_vals):
            if k is not None and k is not self.DELETED:
                self.put(k, v)

def two_sum(nums, target):
    """经典哈希 O(n)"""
    seen = {}
    for i, n in enumerate(nums):
        if target - n in seen:
            return [seen[target - n], i]

def group_anagrams(words):

class BinaryTree:

    # ── 递归版遍历（简洁）──
    def inorder_recursive(self, root):
        if not root: return []
        return (self.inorder_recursive(root.left)
                + [root.val]
                + self.inorder_recursive(root.right))

    # ── 迭代版中序（面试常考）──
    def inorder(self, root):
        result, stack = [], []
        cur = root
        while cur or stack:
            while cur:                 # 一直向左
                stack.append(cur)
                cur = cur.left
            cur = stack.pop()          # 处理节点
            result.append(cur.val)
            cur = cur.right            # 转向右子树
        return result

    # ── 迭代版前序 ──
    def preorder(self, root):
        if not root: return []
        result, stack = [], [root]
        while stack:
            node = stack.pop()
            result.append(node.val)
            if node.right: stack.append(node.right)  # 右先入栈（后处理）
            if node.left:  stack.append(node.left)
        return result

    # ── 迭代版后序（反转前序变体）──
    def postorder(self, root):
        if not root: return []
        result, stack = [], [root]
        while stack:
            node = stack.pop()
            result.append(node.val)
            if node.left:  stack.append(node.left)   # 左先入栈
            if node.right: stack.append(node.right)
        return result[::-1]            # 翻转得后序

    # ── 层序遍历（BFS，按层分组）──
    def levelorder(self, root):
        if not root: return []
        result, q = [], deque([root])
        while q:
            level = []
            for _ in range(len(q)):    # 当前层的节点数
                node = q.popleft()
                level.append(node.val)
                if node.left:  q.append(node.left)
                if node.right: q.append(node.right)
            result.append(level)
        return result

    # ── 树的高度（后序递归）──
    def height(self, root):
        if not root: return 0
        return 1 + max(self.height(root.left), self.height(root.right))

    # ── 判断平衡二叉树 ──
    def is_balanced(self, root):
        def check(node):
            if not node: return 0
            l = check(node.left)
            if l == -1: return -1
            r = check(node.right)
            if r == -1: return -1
            if abs(l - r) > 1: return -1  # 不平衡
            return max(l, r) + 1
        return check(root) != -1

    # ── 最低公共祖先（LCA）──
    def lca(self, root, p, q):
        if not root or root is p or root is q:
            return root
        left  = self.lca(root.left, p, q)
        right = self.lca(root.right, p, q)
        if left and right: return root   # p,q 分别在左右子树
        return left or right

    # ── 路径总和 ──
    def has_path_sum(self, root, target):
        if not root: return False
        if not root.left and not root.right:
            return root.val == target
        return (self.has_path_sum(root.left,  target - root.val) or
                self.has_path_sum(root.right, target - root.val))

class BST:
    def __init__(self): self.root = None

    # ── 插入 ──
    def insert(self, val):
        self.root = self._insert(self.root, val)

    def _insert(self, node, val):
        if not node: return TreeNode(val)
        if val < node.val:   node.left  = self._insert(node.left, val)
        elif val > node.val: node.right = self._insert(node.right, val)
        return node   # 相等时不重复插入

    # ── 搜索 ──
    def search(self, val):
        node = self.root
        while node:
            if val == node.val:   return node
            elif val < node.val:  node = node.left
            else:                 node = node.right
        return None

    # ── 删除（递归，三种情况）──
    def delete(self, val):
        self.root = self._delete(self.root, val)

    def _delete(self, node, val):
        if not node: return None
        if val < node.val:
            node.left = self._delete(node.left, val)
        elif val > node.val:
            node.right = self._delete(node.right, val)
        else:
            # Case 1 & 2: 0 或 1 个子节点
            if not node.left:  return node.right
            if not node.right: return node.left
            # Case 3: 两个子节点 → 找中序后继（右子树最小值）
            successor = self._min_node(node.right)
            node.val = successor.val                 # 覆盖
            node.right = self._delete(node.right, successor.val)
        return node

    def _min_node(self, node):
        while node.left: node = node.left
        return node

    def _max_node(self, node):
        while node.right: node = node.right
        return node

    # ── 验证是否合法 BST ──
    def is_valid_bst(self, root, lo=float('-inf'), hi=float('inf')):
        if not root: return True
        if not lo < root.val < hi: return False
        return (self.is_valid_bst(root.left,  lo, root.val) and
                self.is_valid_bst(root.right, root.val, hi))

    # ── 第 k 小的元素（中序第 k 个）──
    def kth_smallest(self, root, k):
        stack = []
        cur = root
        while cur or stack:
            while cur:
                stack.append(cur)
                cur = cur.left
            cur = stack.pop()
            k -= 1
            if k == 0: return cur.val
            cur = cur.right

    # ── 前驱（inorder predecessor）和后继 ──
    def inorder_successor(self, root, p):
        successor = None
        while root:
            if p.val < root.val:
                successor = root
                root = root.left
            else:
                root = root.right
        return successor

    # ── 范围查询：[lo, hi] 内的所有值 ──
    def range_search(self, root, lo, hi, result=None):
        if result is None: result = []
        if not root: return result
        if lo < root.val:
            self.range_search(root.left, lo, hi, result)
        if lo <= root.val <= hi:
            result.append(root.val)
        if root.val < hi:
            self.range_search(root.right, lo, hi, result)
        return result

class AVLTree:
    def __init__(self): self.root = None

    def _h(self, n): return n.height if n else 0
    def _bf(self, n): return self._h(n.left) - self._h(n.right)  # 平衡因子
    def _update(self, n): n.height = 1 + max(self._h(n.left), self._h(n.right))

    # ── 左旋（RR 情况）──
    def _rotate_left(self, x):
        y, T2 = x.right, x.right.left
        y.left = x; x.right = T2
        self._update(x); self._update(y)
        return y                     # y 成为新根

    # ── 右旋（LL 情况）──
    def _rotate_right(self, y):
        x, T2 = y.left, y.left.right
        x.right = y; y.left = T2
        self._update(y); self._update(x)
        return x

    # ── 再平衡（四种情况）──
    def _rebalance(self, node):
        self._update(node)
        bf = self._bf(node)

        if bf > 1:                   # 左重
            if self._bf(node.left) < 0:          # LR
                node.left = self._rotate_left(node.left)
            return self._rotate_right(node)      # LL（含 LR 处理后）

        if bf < -1:                  # 右重
            if self._bf(node.right) > 0:         # RL
                node.right = self._rotate_right(node.right)
            return self._rotate_left(node)       # RR（含 RL 处理后）

        return node                  # 已平衡

    # ── 插入 ──
    def insert(self, val): self.root = self._insert(self.root, val)

    def _insert(self, node, val):
        if not node: return AVLNode(val)
        if val < node.val:   node.left  = self._insert(node.left, val)
        elif val > node.val: node.right = self._insert(node.right, val)
        else: return node            # 重复值不插入
        return self._rebalance(node)

    # ── 删除 ──
    def delete(self, val): self.root = self._delete(self.root, val)

    def _delete(self, node, val):
        if not node: return None
        if val < node.val:
            node.left = self._delete(node.left, val)
        elif val > node.val:
            node.right = self._delete(node.right, val)
        else:
            if not node.left: return node.right
            if not node.right: return node.left
            # 用中序后继替换
            succ = node.right
            while succ.left: succ = succ.left
            node.val = succ.val
            node.right = self._delete(node.right, succ.val)
        return self._rebalance(node)

    # ── 搜索（与 BST 相同）──
    def search(self, val):
        node = self.root
        while node:
            if val == node.val: return True
            node = node.left if val < node.val else node.right
        return False

class MinHeap:
    """最小堆：手写实现，便于理解"""
    def __init__(self): self.h = []

    def __len__(self): return len(self.h)
    def peek(self):    return self.h[0]

    def push(self, val):
        self.h.append(val)
        self._sift_up(len(self.h) - 1)    # 上浮

    def pop(self):
        self._swap(0, len(self.h) - 1)    # 根与末尾交换
        val = self.h.pop()                 # 移除末尾（原根）
        if self.h: self._sift_down(0)     # 新根下沉
        return val

    def _sift_up(self, i):
        while i > 0:
            p = (i - 1) // 2
            if self.h[i] < self.h[p]:     # 子 < 父 → 交换上浮
                self._swap(i, p); i = p
            else: break

    def _sift_down(self, i):
        n = len(self.h)
        while True:
            s = i                          # smallest
            l, r = 2*i+1, 2*i+2
            if l < n and self.h[l] < self.h[s]: s = l
            if r < n and self.h[r] < self.h[s]: s = r
            if s == i: break
            self._swap(i, s); i = s

    def _swap(self, i, j):
        self.h[i], self.h[j] = self.h[j], self.h[i]

    def heapify(self, arr):
        """O(n) 建堆：从最后非叶节点向前 sift_down"""
        self.h = arr[:]
        for i in range(len(self.h) // 2 - 1, -1, -1):
            self._sift_down(i)

def top_k_largest(nums, k):
    """维护大小为 k 的最小堆，O(n log k)"""
    heap = nums[:k]
    heapq.heapify(heap)                   # O(k) 建堆
    for n in nums[k:]:
        if n > heap[0]:                   # 比堆顶大才入堆
            heapq.heapreplace(heap, n)    # O(log k)
    return sorted(heap, reverse=True)

class Trie:
    def __init__(self): self.root = TrieNode()

    def insert(self, word):
        node = self.root
        for ch in word:
            if ch not in node.children:
                node.children[ch] = TrieNode()
            node = node.children[ch]
            node.count += 1
        node.is_end = True

    def search(self, word):
        """精确匹配"""
        node = self._find(word)
        return node is not None and node.is_end

    def starts_with(self, prefix):
        """前缀存在则返回 True"""
        return self._find(prefix) is not None

    def count_prefix(self, prefix):
        """以 prefix 开头的单词数"""
        node = self._find(prefix)
        return node.count if node else 0

    def _find(self, prefix):
        node = self.root
        for ch in prefix:
            if ch not in node.children: return None
            node = node.children[ch]
        return node

    def delete(self, word):
        """递归删除，若某节点不再需要则移除"""
        def _del(node, word, depth):
            if not node: return False
            if depth == len(word):
                if not node.is_end: return False
                node.is_end = False
                return len(node.children) == 0   # 无子节点可删除
            ch = word[depth]
            if ch not in node.children: return False
            if _del(node.children[ch], word, depth + 1):
                del node.children[ch]
                return not node.is_end and len(node.children) == 0
            return False
        _del(self.root, word, 0)

    def autocomplete(self, prefix):
        """自动补全：返回所有以 prefix 为前缀的单词"""
        node = self._find(prefix)
        if not node: return []
        result = []
        def dfs(n, path):
            if n.is_end: result.append(prefix + ''.join(path))
            for ch, child in sorted(n.children.items()):
                dfs(child, path + [ch])
        dfs(node, [])
        return result

from collections import defaultdict, deque
import heapq

class Graph:
    def __init__(self, directed=False):
        self.adj = defaultdict(list)   # {u: [(v, weight), ...]}
        self.directed = directed

    def add_edge(self, u, v, w=1):
        self.adj[u].append((v, w))
        if not self.directed: self.adj[v].append((u, w))

    # ── BFS：最短路径（无权图）──
    def bfs(self, start):
        visited = {start}
        queue, order = deque([start]), []
        while queue:
            node = queue.popleft()
            order.append(node)
            for nei, _ in self.adj[node]:
                if nei not in visited:
                    visited.add(nei); queue.append(nei)
        return order

    # ── DFS：连通分量、路径检测 ──
    def dfs(self, start):
        visited, order = set(), []
        def _dfs(u):
            visited.add(u); order.append(u)
            for v, _ in self.adj[u]:
                if v not in visited: _dfs(v)
        _dfs(start)
        return order

    # ── Dijkstra：单源最短路径（非负权图）──
    def dijkstra(self, src):
        dist = defaultdict(lambda: float('inf'))
        dist[src] = 0
        pq = [(0, src)]                     # (距离, 节点)
        while pq:
            d, u = heapq.heappop(pq)
            if d > dist[u]: continue        # 已处理过更短路径
            for v, w in self.adj[u]:
                if dist[u] + w < dist[v]:
                    dist[v] = dist[u] + w
                    heapq.heappush(pq, (dist[v], v))
        return dict(dist)

    # ── Bellman-Ford：含负权边（可检测负环）──
    def bellman_ford(self, src, vertices):
        dist = {v: float('inf') for v in vertices}
        dist[src] = 0
        for _ in range(len(vertices) - 1):  # V-1 轮松弛
            for u in self.adj:
                for v, w in self.adj[u]:
                    if dist[u] + w < dist[v]:
                        dist[v] = dist[u] + w
        # 检测负环
        for u in self.adj:
            for v, w in self.adj[u]:
                if dist[u] + w < dist[v]:
                    return None            # 存在负环
        return dist

    # ── 拓扑排序（Kahn，BFS 入度法）──
    def topological_sort(self):
        in_deg = defaultdict(int)
        nodes = set(self.adj.keys())
        for u in self.adj:
            for v, _ in self.adj[u]:
                in_deg[v] += 1
                nodes.add(v)
        queue = deque(n for n in nodes if in_deg[n] == 0)
        order = []
        while queue:
            u = queue.popleft(); order.append(u)
            for v, _ in self.adj[u]:
                in_deg[v] -= 1
                if in_deg[v] == 0: queue.append(v)
        return order if len(order) == len(nodes) else []  # [] 表示有环

    # ── 检测有向图是否有环（DFS 三色标记）──
    def has_cycle_directed(self):
        WHITE, GRAY, BLACK = 0, 1, 2
        color = defaultdict(int)
        def dfs(u):
            color[u] = GRAY
            for v, _ in self.adj[u]:
                if color[v] == GRAY: return True   # 发现后向边
                if color[v] == WHITE and dfs(v): return True
            color[u] = BLACK
            return False
        return any(dfs(u) for u in self.adj if color[u] == WHITE)

    # ── 最小生成树 Prim 算法 ──
    def prim(self, start):
        visited = {start}
        edges = [(w, start, v) for v, w in self.adj[start]]
        heapq.heapify(edges)
        mst_cost, mst_edges = 0, []
        while edges:
            w, u, v = heapq.heappop(edges)
            if v in visited: continue
            visited.add(v)
            mst_cost += w; mst_edges.append((u, v, w))
            for nei, nw in self.adj[v]:
                if nei not in visited:
                    heapq.heappush(edges, (nw, v, nei))
        return mst_cost, mst_edges

class SkipList:
    MAX_LEVEL = 16     # 最大层数，log₂(n) 量级
    P = 0.5            # 每层提升概率

    def __init__(self):
        # 头哨兵：值 -∞，最大层数
        self.head = SkipNode(float('-inf'), self.MAX_LEVEL)
        self.level = 0    # 当前最高层（动态）
        self.size  = 0

    def _random_level(self):
        """随机生成层数，期望值 = 1/(1-P) = 2"""
        lvl = 0
        while random.random() < self.P and lvl < self.MAX_LEVEL:
            lvl += 1
        return lvl

    def search(self, target) -> bool:
        cur = self.head
        # 从最高层向下逐层逼近
        for i in range(self.level, -1, -1):
            while cur.forward[i] and cur.forward[i].val < target:
                cur = cur.forward[i]
        cur = cur.forward[0]
        return cur is not None and cur.val == target

    def insert(self, val):
        # update[i] = 第 i 层中，新节点的前驱节点
        update = [self.head] * (self.MAX_LEVEL + 1)
        cur = self.head

        for i in range(self.level, -1, -1):
            while cur.forward[i] and cur.forward[i].val < val:
                cur = cur.forward[i]
            update[i] = cur

        new_level = self._random_level()
        # 若新层数超出当前最大层，补充 update
        if new_level > self.level:
            for i in range(self.level + 1, new_level + 1):
                update[i] = self.head
            self.level = new_level

        new_node = SkipNode(val, new_level)
        for i in range(new_level + 1):
            new_node.forward[i] = update[i].forward[i]
            update[i].forward[i] = new_node
        self.size += 1

    def delete(self, val) -> bool:
        update = [None] * (self.MAX_LEVEL + 1)
        cur = self.head

        for i in range(self.level, -1, -1):
            while cur.forward[i] and cur.forward[i].val < val:
                cur = cur.forward[i]
            update[i] = cur

        target = cur.forward[0]
        if not target or target.val != val:
            return False   # 不存在

        for i in range(self.level + 1):
            if update[i].forward[i] is not target: break
            update[i].forward[i] = target.forward[i]

        # 缩减 level
        while self.level > 0 and not self.head.forward[self.level]:
            self.level -= 1
        self.size -= 1
        return True

    def range_query(self, lo, hi):
        """范围查询 [lo, hi]，O(log n + k)"""
        cur = self.head
        for i in range(self.level, -1, -1):
            while cur.forward[i] and cur.forward[i].val < lo:
                cur = cur.forward[i]
        cur = cur.forward[0]
        result = []
        while cur and cur.val <= hi:
            result.append(cur.val)
            cur = cur.forward[0]
        return result

    def __repr__(self):
        vals = []
        cur = self.head.forward[0]
        while cur: vals.append(str(cur.val)); cur = cur.forward[0]
        return ' → '.join(vals)

class BPlusTree:
    def __init__(self, order=4):
        self.root = BPNode(is_leaf=True)
        self.order = order          # 每个节点最多 order-1 个键

    # ── 查找 ──
    def search(self, key):
        leaf = self._find_leaf(self.root, key)
        for i, k in enumerate(leaf.keys):
            if k == key: return leaf.values[i]
        return None

    def _find_leaf(self, node, key):
        if node.is_leaf: return node
        for i, k in enumerate(node.keys):
            if key < k: return self._find_leaf(node.children[i], key)
        return self._find_leaf(node.children[-1], key)

    # ── 插入 ──
    def insert(self, key, val=None):
        result = self._insert(self.root, key, val)
        if result:                  # 根节点发生了分裂
            mid_key, new_child = result
            new_root = BPNode(is_leaf=False)
            new_root.keys = [mid_key]
            new_root.children = [self.root, new_child]
            self.root = new_root

    def _insert(self, node, key, val):
        if node.is_leaf:
            # 有序插入
            idx = 0
            while idx < len(node.keys) and node.keys[idx] < key: idx += 1
            node.keys.insert(idx, key)
            node.values.insert(idx, val if val is not None else key)
            if len(node.keys) >= self.order:
                return self._split_leaf(node)
            return None
        else:
            idx = 0
            while idx < len(node.keys) and node.keys[idx] <= key: idx += 1
            result = self._insert(node.children[idx], key, val)
            if result:
                mid_key, new_child = result
                node.keys.insert(idx, mid_key)
                node.children.insert(idx + 1, new_child)
                if len(node.keys) >= self.order:
                    return self._split_internal(node)
            return None

    def _split_leaf(self, node):
        """叶节点分裂：中间键上推（并保留在右叶子中）"""
        mid = len(node.keys) // 2
        new_node = BPNode(is_leaf=True)
        new_node.keys   = node.keys[mid:]
        new_node.values = node.values[mid:]
        new_node.next   = node.next       # 维护叶子链表
        node.next       = new_node
        node.keys   = node.keys[:mid]
        node.values = node.values[:mid]
        return new_node.keys[0], new_node  # 中间键上推

    def _split_internal(self, node):
        """内部节点分裂：中间键上推（不保留在子节点）"""
        mid = len(node.keys) // 2
        mid_key = node.keys[mid]
        new_node = BPNode(is_leaf=False)
        new_node.keys     = node.keys[mid+1:]
        new_node.children = node.children[mid+1:]
        node.keys     = node.keys[:mid]
        node.children = node.children[:mid+1]
        return mid_key, new_node

    # ── 范围查询（B+树最大优势）──
    def range_query(self, lo, hi):
        """O(log n + k)：定位首叶，沿链表扫描"""
        leaf = self._find_leaf(self.root, lo)
        result = []
        while leaf:
            for i, k in enumerate(leaf.keys):
                if lo <= k <= hi: result.append((k, leaf.values[i]))
                elif k > hi:      return result
            leaf = leaf.next
        return result

    def scan_all(self):

        while not node.is_leaf: node = node.children[0]
        result = []
        while node:
            result.extend(zip(node.keys, node.values))

class LRUCache:
    def __init__(self, capacity: int):
        self.cap = capacity
        self.cache = OrderedDict()     # 有序字典，维护插入/访问顺序

    def get(self, key: int) -> int:
        if key not in self.cache: return -1
        self.cache.move_to_end(key)    # 移到尾部（标记最近使用）
        return self.cache[key]

    def put(self, key: int, value: int) -> None:
        if key in self.cache:
            self.cache.move_to_end(key)
        self.cache[key] = value
        if len(self.cache) > self.cap:
            self.cache.popitem(last=False)  # 移除头部（最久未使用）

from collections import defaultdict

class LFUCache:
    def __init__(self, capacity):
        self.cap = capacity
        self.min_freq = 0
        self.key_val  = {}           # key → val
        self.key_freq = {}           # key → 频次
        self.freq_keys = defaultdict(OrderedDict)   # 频次 → OrderedDict(key)

    def get(self, key):
        if key not in self.key_val: return -1
        self._increment(key)
        return self.key_val[key]

    def put(self, key, val):
        if self.cap <= 0: return
        if key in self.key_val:
            self.key_val[key] = val
            self._increment(key)
        else:
            if len(self.key_val) >= self.cap:
                # 淘汰最低频且最久未用的
                evict_key, _ = self.freq_keys[self.min_freq].popitem(last=False)
                del self.key_val[evict_key]; del self.key_freq[evict_key]
            self.key_val[key] = val
            self.key_freq[key] = 1
            self.freq_keys[1][key] = None
            self.min_freq = 1

    def _increment(self, key):
        f = self.key_freq[key]
        self.key_freq[key] = f + 1
        self.freq_keys[f].pop(key)
        if not self.freq_keys[f] and f == self.min_freq:
            self.min_freq += 1
        self.freq_keys[f + 1][key] = None

class SegTree:
    def __init__(self, nums):
        self.n = len(nums)
        self.tree = [0] * (4 * self.n)
        if nums: self._build(nums, 0, 0, self.n - 1)

    def _build(self, nums, node, s, e):
        if s == e:
            self.tree[node] = nums[s]; return
        mid = (s + e) >> 1
        self._build(nums, 2*node+1, s,   mid)
        self._build(nums, 2*node+2, mid+1, e)
        self.tree[node] = self.tree[2*node+1] + self.tree[2*node+2]

    def update(self, idx, val, node=0, s=0, e=None):
        """单点更新：将 nums[idx] 设为 val"""
        if e is None: e = self.n - 1
        if s == e:
            self.tree[node] = val; return
        mid = (s + e) >> 1
        if idx <= mid: self.update(idx, val, 2*node+1, s, mid)
        else:          self.update(idx, val, 2*node+2, mid+1, e)
        self.tree[node] = self.tree[2*node+1] + self.tree[2*node+2]

    def query(self, l, r, node=0, s=0, e=None):
        """区间查询 [l, r] 的和"""
        if e is None: e = self.n - 1
        if r < s or e < l: return 0           # 完全不相交
        if l <= s and e <= r: return self.tree[node]  # 完全覆盖
        mid = (s + e) >> 1
        return (self.query(l, r, 2*node+1, s, mid) +
                self.query(l, r, 2*node+2, mid+1, e))

class LazySegTree:
    def __init__(self, n):
        self.n = n
        self.tree = [0] * (4 * n)   # 区间和
        self.lazy = [0] * (4 * n)   # 懒标记（待下推的增量）

    def _push_down(self, node, s, e):
        """将懒标记下推给子节点"""
        if self.lazy[node] == 0: return
        mid = (s + e) >> 1
        lc, rc = 2*node+1, 2*node+2
        self.tree[lc] += self.lazy[node] * (mid - s + 1)
        self.tree[rc] += self.lazy[node] * (e - mid)
        self.lazy[lc] += self.lazy[node]
        self.lazy[rc] += self.lazy[node]
        self.lazy[node] = 0

    def range_add(self, l, r, val, node=0, s=0, e=None):
        """区间 [l,r] 每个元素 +val"""
        if e is None: e = self.n - 1
        if r < s or e < l: return
        if l <= s and e <= r:
            self.tree[node] += val * (e - s + 1)
            self.lazy[node] += val
            return
        self._push_down(node, s, e)
        mid = (s + e) >> 1
        self.range_add(l, r, val, 2*node+1, s, mid)
        self.range_add(l, r, val, 2*node+2, mid+1, e)
        self.tree[node] = self.tree[2*node+1] + self.tree[2*node+2]

    def query(self, l, r, node=0, s=0, e=None):
        if e is None: e = self.n - 1
        if r < s or e < l: return 0
        if l <= s and e <= r: return self.tree[node]
        self._push_down(node, s, e)
        mid = (s + e) >> 1
        return (self.query(l, r, 2*node+1, s, mid) +
                self.query(l, r, 2*node+2, mid+1, e))

class DynamicSegNode:
    __slots__ = ['val', 'lazy', 'left', 'right']

class DynamicSegTree:
    def __init__(self, lo, hi):
        self.root = DynamicSegNode()
        self.lo, self.hi = lo, hi

    def update(self, l, r, val, node=None, s=None, e=None):
        if node is None: node, s, e = self.root, self.lo, self.hi
        if r < s or e < l: return
        if l <= s and e <= r:
            node.val += val * (e - s + 1); node.lazy += val; return
        if not node.left:  node.left  = DynamicSegNode()
        if not node.right: node.right = DynamicSegNode()
        if node.lazy:      # 推下懒标记
            mid = (s + e) >> 1
            for c, sz in [(node.left, mid-s+1), (node.right, e-mid)]:
                c.val += node.lazy * sz; c.lazy += node.lazy
            node.lazy = 0
        mid = (s + e) >> 1
        self.update(l, r, val, node.left,  s, mid)
        self.update(l, r, val, node.right, mid+1, e)
        node.val = ((node.left.val  if node.left  else 0) +
                    (node.right.val if node.right else 0))

# ─── 基础树状数组（单点更新 + 前缀和）1-indexed ───
class BIT:
    def __init__(self, n):
        self.n = n
        self.tree = [0] * (n + 1)   # 1-indexed

    def update(self, i, delta):
        """将第 i 个位置加上 delta"""
        while i <= self.n:
            self.tree[i] += delta
            i += i & (-i)           # 向上传递

    def prefix(self, i):
        """前 i 个元素之和"""
        s = 0
        while i > 0:
            s += self.tree[i]
            i -= i & (-i)           # 去掉最低位 1
        return s

    def query(self, l, r):          # 区间 [l,r] 之和
        return self.prefix(r) - self.prefix(l - 1)

    def build(self, arr):           # O(n) 建树
        for i, v in enumerate(arr, 1):
            self.update(i, v)

    def find_kth(self, k):
        """二分查找第 k 小，O(log n)"""
        pos = 0
        bit_mask = 1 << (self.n.bit_length() - 1)
        while bit_mask:
            nxt = pos + bit_mask
            if nxt <= self.n and self.tree[nxt] < k:
                k -= self.tree[nxt]
                pos = nxt
            bit_mask >>= 1
        return pos + 1

class BIT2D:

        self.tree = [[0] * (n + 1) for _ in range(m + 1)]

    def update(self, x, y, delta):
        i = x
        while i <= self.m:
            j = y
            while j <= self.n:
                self.tree[i][j] += delta
                j += j & (-j)
            i += i & (-i)

    def prefix(self, x, y):
        s, i = 0, x
        while i > 0:
            j = y
            while j > 0:
                s += self.tree[i][j]
                j -= j & (-j)
            i -= i & (-i)
        return s

    def query(self, x1, y1, x2, y2):
        return (self.prefix(x2, y2)
              - self.prefix(x1-1, y2)
              - self.prefix(x2, y1-1)
              + self.prefix(x1-1, y1-1))

class UnionFind:
    def __init__(self, n):
        self.parent = list(range(n))   # parent[i] = i（初始：每个节点是自己的根）
        self.rank   = [0] * n          # 树高上界（用于按秩合并）
        self.size   = [1] * n          # 集合大小
        self.count  = n                # 连通分量数

    def find(self, x) -> int:
        """路径压缩：递归版（find 后节点直接指向根）"""
        if self.parent[x] != x:
            self.parent[x] = self.find(self.parent[x])
        return self.parent[x]

    def find_iterative(self, x) -> int:
        """路径压缩：迭代版（两趟：先找根，再压缩）"""
        root = x
        while self.parent[root] != root:
            root = self.parent[root]
        while self.parent[x] != root:
            self.parent[x], x = root, self.parent[x]
        return root

    def union(self, x, y) -> bool:
        """按秩合并：矮树指向高树（保持低树高）"""
        px, py = self.find(x), self.find(y)
        if px == py: return False       # 已在同一集合
        if self.rank[px] < self.rank[py]:
            px, py = py, px             # 确保 px 是更高的根
        self.parent[py] = px            # py 指向 px
        self.size[px] += self.size[py]
        if self.rank[px] == self.rank[py]:
            self.rank[px] += 1          # 等高时合并，高度+1
        self.count -= 1
        return True

    def connected(self, x, y) -> bool:
        return self.find(x) == self.find(y)

    def get_size(self, x) -> int:
        return self.size[self.find(x)]

import math

class SparseTable:
    """静态区间最小值（RMQ），O(n log n) 预处理，O(1) 查询"""

    def __init__(self, nums, func=min):
        """
        func: 幂等操作，可为 min / max / gcd（不能是 sum！）
        """
        self.func = func
        n = len(nums)
        self.LOG = max(1, int(math.log2(n)) + 1)
        # st[j][i] = 从 i 开始长度 2^j 的区间的 func 值
        self.st = [[0] * n for _ in range(self.LOG)]
        self.st[0] = nums[:]

        # 倍增填表
        for j in range(1, self.LOG):
            for i in range(n - (1 << j) + 1):
                self.st[j][i] = func(
                    self.st[j-1][i],
                    self.st[j-1][i + (1 << (j-1))]
                )

        # 预计算 log2（避免查询时重复计算）
        self.log2 = [0] * (n + 1)
        for i in range(2, n + 1):
            self.log2[i] = self.log2[i >> 1] + 1

    def query(self, l, r) -> int:
        """O(1) 区间查询 [l, r]"""
        k = self.log2[r - l + 1]
        return self.func(self.st[k][l], self.st[k][r - (1 << k) + 1])

        self.euler = []      # 欧拉序
        self.depth = []      # 对应深度
        self.first = {}      # 节点首次出现位置

        def dfs(node, d):
            if not node: return
            self.first[node.val] = len(self.euler)
            self.euler.append(node.val)
            self.depth.append(d)
            if node.left:
                dfs(node.left, d + 1)
                self.euler.append(node.val)
                self.depth.append(d)
            if node.right:
                dfs(node.right, d + 1)
                self.euler.append(node.val)
                self.depth.append(d)

        dfs(root, 0)
        # Sparse Table 基于深度数组
        self.st = SparseTable(self.depth)

    def lca(self, u, v):
        l, r = self.first[u], self.first[v]
        if l > r: l, r = r, l
        min_depth = self.st.query(l, r)
        # 找欧拉序中对应最小深度的节点
        for i in range(l, r + 1):
            if self.depth[i] == min_depth:
                return self.euler[i]

# ─── 字符串哈希（单哈希）───
class StringHash:
    BASE = 131
    MOD  = (1 << 61) - 1    # 梅森质数，碰撞率极低

    def __init__(self, s: str):
        n = len(s)
        self.n = n
        self.h = [0] * (n + 1)    # 前缀哈希
        self.pw = [1] * (n + 1)   # BASE 的幂次

        for i, ch in enumerate(s):
            self.h[i+1]  = (self.h[i] * self.BASE + ord(ch)) % self.MOD
            self.pw[i+1] = (self.pw[i] * self.BASE) % self.MOD

    def get(self, l: int, r: int) -> int:
        """子串 s[l..r]（闭区间）的哈希值，O(1)"""
        return (self.h[r+1] - self.h[l] * self.pw[r-l+1]) % self.MOD

    def equal(self, l1, r1, l2, r2) -> bool:
        """比较两个子串是否相等，O(1)"""
        return self.get(l1, r1) == self.get(l2, r2)

class DoubleHash:
    B1, M1 = 131, (1 << 61) - 1
    B2, M2 = 137, (1 << 31) - 1

    def __init__(self, s: str):
        n = len(s)
        self.h1 = [0] * (n+1); self.p1 = [1] * (n+1)
        self.h2 = [0] * (n+1); self.p2 = [1] * (n+1)
        for i, ch in enumerate(s):
            v = ord(ch)
            self.h1[i+1] = (self.h1[i] * self.B1 + v) % self.M1
            self.h2[i+1] = (self.h2[i] * self.B2 + v) % self.M2
            self.p1[i+1] = self.p1[i] * self.B1 % self.M1
            self.p2[i+1] = self.p2[i] * self.B2 % self.M2

    def get(self, l, r):
        h1 = (self.h1[r+1] - self.h1[l] * self.p1[r-l+1]) % self.M1
        h2 = (self.h2[r+1] - self.h2[l] * self.p2[r-l+1]) % self.M2
        return (h1, h2)

    def equal(self, l1, r1, l2, r2):
        return self.get(l1, r1) == self.get(l2, r2)


# ─── Rabin-Karp 字符串匹配 O(n+m) 期望 ───
def rabin_karp(text: str, pattern: str):
    """返回 pattern 在 text 中所有出现位置"""
    n, m = len(text), len(pattern)
    if m > n: return []

    BASE, MOD = 131, (1 << 61) - 1
    # 计算 pattern 的哈希
    ph = 0
    for ch in pattern:
        ph = (ph * BASE + ord(ch)) % MOD

    # 滚动哈希遍历 text
    th = 0
    high = pow(BASE, m, MOD)    # BASE^m
    result = []

    for i, ch in enumerate(text):
        th = (th * BASE + ord(ch)) % MOD
        if i >= m:                               # 滑出窗口
            th = (th - ord(text[i-m]) * high) % MOD
        if i >= m - 1 and th == ph:              # 哈希匹配
            if text[i-m+1:i+1] == pattern:      # 验证（防碰撞）
                result.append(i - m + 1)
    return result


# ─── 应用：最长公共子串（二分 + 哈希，O(n log n)）───
def longest_common_substring(s: str, t: str) -> str:
    def has_common(length):
        seen = set()
        sh = StringHash(s)
        for i in range(len(s) - length + 1):
            seen.add(sh.get(i, i + length - 1))
        th = StringHash(t)
        for i in range(len(t) - length + 1):
            if th.get(i, i + length - 1) in seen:
                return i
        return -1

    lo, hi, best_start, best_len = 0, min(len(s), len(t)), 0, 0
    while lo <= hi:
        mid = (lo + hi) // 2
        pos = has_common(mid)
        if pos != -1:
            best_len, best_start = mid, pos
            lo = mid + 1
        else:
            hi = mid - 1
    return t[best_start:best_start + best_len]


# ─── 应用：最长回文子串（Manacher 算法，O(n)）───
def manacher(s: str) -> str:
    """Manacher 算法：O(n) 最长回文子串"""
    # 插入分隔符处理奇偶统一
    t = '#' + '#'.join(s) + '#'
    n = len(t)
    p = [0] * n       # p[i] = 以 t[i] 为中心的回文半径
    c = r = 0          # 当前最右回文的中心和右边界

    for i in range(n):
        if i < r:
            p[i] = min(r - i, p[2*c - i])   # 利用已知信息
        while (i - p[i] - 1 >= 0 and i + p[i] + 1 < n
               and t[i - p[i] - 1] == t[i + p[i] + 1]):
            p[i] += 1
        if i + p[i] > r:
            c, r = i, i + p[i]

    max_len = max(p)
    center  = p.index(max_len)
    start   = (center - max_len) // 2
    return s[start:start + max_len]

# ══════════════════ KMP 算法 ══════════════════

def build_pi(pattern: str) -> list:
    """
    构建失配函数（π 数组 / next 数组）O(m)
    π[i] = pattern[0..i] 的最长真前缀 = 真后缀的长度
    """
    m = len(pattern)
    pi = [0] * m
    k = 0               # 当前匹配的前缀长度
    for i in range(1, m):
        while k > 0 and pattern[k] != pattern[i]:
            k = pi[k-1]    # 失配：退到上一个可能的前缀
        if pattern[k] == pattern[i]:
            k += 1
        pi[i] = k
    return pi


def kmp_search(text: str, pattern: str) -> list:
    """KMP 字符串匹配，O(n+m)，返回所有匹配起始下标"""
    if not pattern: return []
    pi = build_pi(pattern)
    n, m = len(text), len(pattern)
    result = []
    k = 0               # 已匹配 pattern 的前 k 个字符

    for i, ch in enumerate(text):
        while k > 0 and pattern[k] != ch:
            k = pi[k-1]    # 利用 π 避免回退到 0
        if pattern[k] == ch:
            k += 1
        if k == m:         # 完整匹配
            result.append(i - m + 1)
            k = pi[k-1]    # 继续搜索下一个可能的匹配
    return result


def kmp_count_occurrences(text: str, pattern: str) -> int:
    """统计 pattern 在 text 中出现次数（含重叠）"""
    return len(kmp_search(text, pattern))


def min_repeat_unit(s: str) -> str:
    """
    最小循环节：KMP 经典应用
    若 len(s) % (len(s) - π[-1]) == 0，则循环节为 s[: len(s)-π[-1]]
    """
    pi = build_pi(s)
    unit_len = len(s) - pi[-1]
    if len(s) % unit_len == 0:
        return s[:unit_len]
    return s   # 不能被整除则自身就是最小循环节


# ══════════════════ Z 函数 ══════════════════

def build_z(s: str) -> list:
    """
    构建 Z 数组，O(n)
    Z[i] = s[i:] 与 s 的最长公共前缀长度（Z[0]=0 或 n，约定为 0）
    """
    n = len(s)
    z = [0] * n
    l = r = 0           # 当前最右 Z-box [l, r]
    for i in range(1, n):
        if i < r:
            z[i] = min(r - i, z[i - l])   # 利用已知的 Z-box
        while i + z[i] < n and s[z[i]] == s[i + z[i]]:
            z[i] += 1
        if i + z[i] > r:
            l, r = i, i + z[i]
    return z


def z_search(text: str, pattern: str) -> list:
    """用 Z 函数做字符串匹配，O(n+m)"""
    m = len(pattern)
    s = pattern + '$' + text    # '$' 是不出现在文本中的分隔符
    z = build_z(s)
    result = []
    for i in range(m + 1, len(s)):
        if z[i] == m:
            result.append(i - m - 1)   # 还原到 text 中的位置
    return result

def count_prefix_occurrences(s: str) -> list:
    """
    Z 经典应用：Z[i] 即为以位置 i 开始的前缀在 s 中出现次数的贡献
    返回每个前缀长度在 s 中出现的总次数
    """
    z = build_z(s)
    n = len(s)
    cnt = [0] * (n + 1)
    cnt[n] += 1            # 前缀 s 本身
    for i in range(1, n):
        cnt[z[i]] += 1
    # 后缀和：长度 k 的前缀出现次数 = sum(cnt[k..n])
    for i in range(n - 1, 0, -1):
        cnt[i] += cnt[i+1]
    return cnt[1:]         # 下标 0 表示长度 1 的前缀

import math
import mmh3          # pip install mmh3（MurmurHash3，工业级非加密哈希）
# 若没有 mmh3，使用下方纯 Python 实现

class BloomFilter:
    def __init__(self, n: int, fp_rate: float = 0.01):
        """
        n:       预期元素数量
        fp_rate: 可接受的假阳性率（默认 1%）
        """
        self.n       = n
        self.fp_rate = fp_rate
        # 最优参数计算
        self.m = self._optimal_m(n, fp_rate)   # 位数组大小
        self.k = self._optimal_k(self.m, n)    # 哈希函数数
        self.bit_array = bytearray(math.ceil(self.m / 8))
        self.count = 0

    @staticmethod
    def _optimal_m(n, p):
        return int(-n * math.log(p) / (math.log(2) ** 2))

    @staticmethod
    def _optimal_k(m, n):
        return max(1, round(m / n * math.log(2)))

    def _hashes(self, item):
        """用 mmh3 的不同种子模拟 k 个独立哈希函数"""
        item_bytes = str(item).encode()
        for seed in range(self.k):
            yield mmh3.hash(item_bytes, seed) % self.m

    def _get_bit(self, idx):
        return bool(self.bit_array[idx // 8] & (1 << (idx % 8)))

    def _set_bit(self, idx):
        self.bit_array[idx // 8] |= (1 << (idx % 8))

    def add(self, item):
        for h in self._hashes(item):
            self._set_bit(h)
        self.count += 1

    def contains(self, item) -> bool:
        """
        False → 元素一定不存在（无假阴性）
        True  → 元素可能存在（有假阳性）
        """
        return all(self._get_bit(h) for h in self._hashes(item))

    @property
    def current_fp_rate(self) -> float:
        """当前实际假阳性率"""
        return (1 - math.exp(-self.k * self.count / self.m)) ** self.k

class BloomFilterPure:
    def __init__(self, n, fp_rate=0.01):
        self.m = int(-n * math.log(fp_rate) / (math.log(2)**2))
        self.k = max(1, round(self.m / n * math.log(2)))
        self.bits = bytearray(math.ceil(self.m / 8))

    def _hashes(self, item):
        s = str(item).encode()
        h1 = int.from_bytes(__import__('hashlib').md5(s).digest()[:8],  'big')
        h2 = int.from_bytes(__import__('hashlib').sha1(s).digest()[:8], 'big')
        for i in range(self.k):
            yield (h1 + i * h2) % self.m    # 双哈希法模拟多哈希

    def add(self, item):
        for h in self._hashes(item):
            self.bits[h // 8] |= 1 << (h % 8)

    def contains(self, item):
        return all((self.bits[h // 8] >> (h % 8)) & 1 for h in self._hashes(item))

class CountingBloomFilter:
    """每个 bit 替换为计数器，支持删除（但空间多 4x）"""
    def __init__(self, n, fp_rate=0.01):
        self.m = int(-n * math.log(fp_rate) / (math.log(2)**2))
        self.k = max(1, round(self.m / n * math.log(2)))
        self.counters = [0] * self.m

    def _hashes(self, item):
        s = str(item).encode()
        h1 = hash(s) & 0xffffffff
        h2 = hash(s[::-1]) & 0xffffffff
        for i in range(self.k):
            yield (h1 + i * h2) % self.m

    def add(self, item):
        for h in self._hashes(item): self.counters[h] += 1

    def remove(self, item):
        if not self.contains(item): raise KeyError("元素不存在")
        for h in self._hashes(item): self.counters[h] -= 1

    def contains(self, item):
        return all(self.counters[h] > 0 for h in self._hashes(item))

def count_ones(n: int) -> int:
    """统计二进制中 1 的个数（Brian Kernighan 算法）O(k)"""
    count = 0
    while n:
        n &= n - 1    # 消去最低位的 1
        count += 1
    return count

def is_power_of_2(n: int) -> bool:

def lowest_bit(n: int) -> int:

def single_number(nums) -> int:
    """只出现一次的数（其余出现两次），XOR O(n)"""
    return __import__('functools').reduce(lambda a,b: a^b, nums)

def single_number_iii(nums):
    """两个只出现一次的数，其余出现两次"""
    xor = 0
    for n in nums: xor ^= n
    # xor = a ^ b，找一个不同的位（如最低位）
    diff_bit = xor & (-xor)
    a = 0
    for n in nums:
        if n & diff_bit:    # 按此位分成两组，各组再 XOR
            a ^= n
    return a, xor ^ a

def missing_number(nums) -> int:
    """0..n 中缺失的数"""
    xor = len(nums)
    for i, n in enumerate(nums):
        xor ^= i ^ n
    return xor

def reverse_bits(n: int, bits=32) -> int:
    """翻转 bits 位整数的二进制"""
    result = 0
    for _ in range(bits):
        result = (result << 1) | (n & 1)
        n >>= 1
    return result

def traveling_salesman(dist):
    """
    旅行商问题（状态压缩 DP），O(2^n · n²)
    dist[i][j] = 从 i 到 j 的距离
    state 的第 i 位 = 是否访问过城市 i
    """
    n = len(dist)
    INF = float('inf')
    # dp[state][i] = 已访问 state 中的城市，当前在城市 i，的最短路径
    dp = [[INF] * n for _ in range(1 << n)]
    dp[1][0] = 0    # 初始：只访问了城市 0，在城市 0

    for state in range(1, 1 << n):
        for u in range(n):
            if not (state >> u & 1): continue       # u 不在已访问集合中
            if dp[state][u] == INF: continue
            for v in range(n):
                if state >> v & 1: continue         # v 已访问
                if dist[u][v] == INF: continue
                new_state = state | (1 << v)
                dp[new_state][v] = min(dp[new_state][v], dp[state][u] + dist[u][v])

    full = (1 << n) - 1
    return min(dp[full][i] + dist[i][0] for i in range(1, n))

def enumerate_subsets(s: int):
    """枚举 s 的所有非空子集，O(2^k) k=popcount(s)"""
    sub = s
    while sub:
        yield sub
        sub = (sub - 1) & s

def enumerate_all_subsets(n: int):
    """枚举 [0, n) 的所有大小为 k 的子集（Gosper's Hack）"""
    def next_same_popcount(x):
        c = x & -x
        r = x + c
        return (((r ^ x) >> 2) // c) | r
    # 示例：n=4, k=2 → 所有含2个元素的子集
    def gen(n, k):
        x = (1 << k) - 1
        while x < (1 << n):
            yield x
            x = next_same_popcount(x)
    return gen

class Bitmap:

单调结构是解决"前/后第一个更大/更小元素"、滑动窗口最值、柱形图面积等问题的利器。全部 O(n)，每个元素入栈/队一次、出栈/队一次，摊还 O(1)/元素。

四种基本问题
  • 下一个更大元素（NGE）：单调递减栈，遇到更大值时弹出
  • 上一个更大元素（PGE）：维护单调递减栈，入栈前栈顶即为答案
  • 下一个更小元素（NSE）：单调递增栈
  • 滑动窗口最大值：单调递减队列，维护窗口内下标
贡献法思路
子数组最小值之和（贡献法）： 对每个 arr[i]，计算其作为最小值的子数组个数 left[i] = 左边第一个 ≤ arr[i] 的位置（下一个更小，左侧） right[i]= 右边第一个 < arr[i] 的位置（右侧严格更小） arr[i] 作为最小值的子数组数量： (i - left[i]) × (right[i] - i) 左侧 ≤（非严格），右侧 <（严格） → 处理重复元素，避免重复计数

def nge(nums):
    """Next Greater Element：下一个更大元素，O(n)"""
    n = len(nums)
    result = [-1] * n
    stack = []                      # 存下标，维护单调递减
    for i, v in enumerate(nums):
        while stack and nums[stack[-1]] < v:
            result[stack.pop()] = v
        stack.append(i)
    return result


def prev_greater(nums):
    """Previous Greater Element：上一个更大元素，O(n)"""
    n = len(nums)
    result = [-1] * n
    stack = []
    for i, v in enumerate(nums):
        while stack and nums[stack[-1]] <= v:
            stack.pop()
        if stack: result[i] = nums[stack[-1]]
        stack.append(i)
    return result


def daily_temperatures(temperatures):
    """LC 739：每天需等几天才能等到更暖的天 O(n)"""
    n = len(temperatures)
    answer = [0] * n
    stack = []
    for i, t in enumerate(temperatures):
        while stack and temperatures[stack[-1]] < t:
            j = stack.pop()
            answer[j] = i - j
        stack.append(i)
    return answer

def largest_rectangle_in_histogram(heights):
    """LC 84：柱状图中最大的矩形，O(n)"""
    heights = [0] + heights + [0]   # 两端加哨兵（高度0）
    stack = [0]                      # 单调递增栈（存下标）
    max_area = 0

    for i in range(1, len(heights)):
        while heights[stack[-1]] > heights[i]:
            h = heights[stack.pop()]
            w = i - stack[-1] - 1   # 宽度 = 右边界 - 左边界 - 1
            max_area = max(max_area, h * w)
        stack.append(i)
    return max_area


def maximal_rectangle(matrix):
    """LC 85：最大矩形（二维扩展），O(m·n)"""
    if not matrix: return 0
    n = len(matrix[0])
    heights = [0] * n
    max_area = 0
    for row in matrix:
        for j in range(n):
            # 更新高度：连续的'1'才算
            heights[j] = heights[j] + 1 if row[j] == '1' else 0
        max_area = max(max_area, largest_rectangle_in_histogram(heights))
    return max_area

def sliding_window_max(nums, k):
    """LC 239：滑动窗口最大值，O(n)"""
    dq = deque()     # 单调递减，存下标
    result = []
    for i, v in enumerate(nums):
        # 移除超出窗口的元素
        if dq and dq[0] <= i - k:
            dq.popleft()
        # 维护单调性
        while dq and nums[dq[-1]] <= v:
            dq.pop()
        dq.append(i)
        if i >= k - 1:
            result.append(nums[dq[0]])
    return result


def min_window_sum_ge_target(nums, target):
    """LC 209：和 ≥ target 的最短子数组（双指针，O(n)）"""
    lo, cur, best = 0, 0, float('inf')
    for hi, v in enumerate(nums):
        cur += v
        while cur >= target:
            best = min(best, hi - lo + 1)
            cur -= nums[lo]; lo += 1
    return 0 if best == float('inf') else best

def sum_subarray_mins(arr):
    """LC 907：子数组最小值之和，贡献法 O(n)"""
    MOD = 10**9 + 7
    n = len(arr)
    left  = [0] * n    # left[i]  = i 到左边第一个 ≤ arr[i] 的距离
    right = [0] * n    # right[i] = i 到右边第一个 <  arr[i] 的距离

    # 计算 left（单调递增栈）
    stack = []
    for i in range(n):
        while stack and arr[stack[-1]] >= arr[i]:
            stack.pop()
        left[i] = i - stack[-1] if stack else i + 1
        stack.append(i)

    # 计算 right
    stack = []
    for i in range(n - 1, -1, -1):
        while stack and arr[stack[-1]] > arr[i]:
            stack.pop()
        right[i] = stack[-1] - i if stack else n - i
        stack.append(i)

    return sum(arr[i] * left[i] * right[i] for i in range(n)) % MOD

def remove_duplicate_letters(s):
    """LC 316 / 1081：去除重复字母，结果字典序最小，O(n)"""
    count = {}
    for c in s: count[c] = count.get(c, 0) + 1

    stack = []
    in_stack = set()

    for c in s:
        count[c] -= 1
        if c in in_stack: continue
        # 栈顶字母 > 当前字母 且 后面还有栈顶字母 → 弹出（单调递增栈）
        while stack and stack[-1] > c and count[stack[-1]] > 0:
            in_stack.remove(stack.pop())
        stack.append(c)
        in_stack.add(c)

    return ''.join(stack)
