括号匹配（栈经典）
遇左括号入栈；遇右括号检查栈顶是否匹配，不匹配或栈空则非法。最终栈为空才合法。
def isValid(s):
    stack = []
    pair = {')': '(', ']': '[', '}': '{'}
    for c in s:
        if c in '([{':
            stack.append(c)
        else:
            if not stack or stack[-1] != pair[c]:
                return False
            stack.pop()
    return not stack
单调栈 — 下一个更大元素
栈内保持单调递减（存下标）。新元素入栈前，弹出所有比它小的元素——"被弹出时就找到了右边第一个更大值"。
def nextGreaterElement(nums):
    n = len(nums)
    ans = [-1] * n
    stack = []          # 存下标，栈内对应值单调递减
    for i in range(n):
        while stack and nums[stack[-1]] < nums[i]:
            idx = stack.pop()
            ans[idx] = nums[i]   # i 就是 idx 右边第一个更大元素
        stack.append(i)
    return ans

# 变体：循环数组 → 遍历 2n，下标用 i % n 取值
单调栈 — 柱状图最大矩形
维护单调递增栈。弹出时以弹出柱为高度，左右边界分别是栈顶和当前 i，计算面积。头尾各加 0 做哨兵，保证最终清空栈。
def largestRectangleArea(heights):
    heights = [0] + heights + [0]   # 哨兵
    stack = [0]
    ans = 0
    for i in range(1, len(heights)):
        while heights[stack[-1]] > heights[i]:
            h = heights[stack.pop()]
            w = i - stack[-1] - 1
            ans = max(ans, h * w)
        stack.append(i)
    return ans
单调队列 — 滑动窗口最大值
双端队列存下标，对应值保持单调递减。队首始终是窗口最大值；入队前弹出所有比新元素小的（它们永远不会是最大值）。
from collections import deque

def maxSlidingWindow(nums, k):
    dq = deque()    # 存下标，对应值单调递减
    ans = []
    for i, x in enumerate(nums):
        # 弹出窗口外的过期元素
        if dq and dq[0] < i - k + 1:
            dq.popleft()
        # 弹出所有比当前值小的（不可能再成为最大值）
        while dq and nums[dq[-1]] < x:
            dq.pop()
        dq.append(i)
        if i >= k - 1:
            ans.append(nums[dq[0]])
    return ans

回溯（Backtracking）

子集 · 组合 · 排列 · 剪枝去重
本质
DFS 暴力枚举 + 剪枝
三类题型
子集 / 组合（无序）/ 排列（有序）
时间
指数级，靠剪枝降常数
回溯 = 做选择 → 递归 → 撤销选择。搞清楚三件事：终止条件、选择列表（从 start 还是全量）、是否需要去重。
通用框架
result = []

def backtrack(path, start):
    if 终止条件:

        return
    for i in range(start, len(choices)):
        if 剪枝条件: continue

子集（无重复元素）
def subsets(nums):
    res = []
    def bt(start, path):
        res.append(path[:])            # 每个节点都是合法子集
        for i in range(start, len(nums)):
            path.append(nums[i])
            bt(i + 1, path)
            path.pop()
    bt(0, [])
    return res
组合总和（元素可重复使用）
def combinationSum(candidates, target):
    candidates.sort()
    res = []
    def bt(start, path, remain):
        if remain == 0:
            res.append(path[:]); return
        for i in range(start, len(candidates)):
            if candidates[i] > remain: break    # 排序后可直接剪枝
            path.append(candidates[i])
            bt(i, path, remain - candidates[i]) # i 不加 1，允许重复用
            path.pop()
    bt(0, [], target)
    return res
排列（含重复元素，去重）
去重关键：同一层不能选相同的值。条件：nums[i]==nums[i-1] 且 used[i-1]==False（前一个同值在本层未被用过）。
def permuteUnique(nums):
    nums.sort()
    res, used = [], [False] * len(nums)
    def bt(path):
        if len(path) == len(nums):
            res.append(path[:]); return
        for i in range(len(nums)):
            if used[i]: continue
            if i > 0 and nums[i] == nums[i-1] and not used[i-1]:
                continue                        # 同层去重
            used[i] = True
            path.append(nums[i])
            bt(path)
            path.pop()
            used[i] = False
    bt([])
    return res

建图 + BFS 最短路（无权图）
BFS 天然求最短路（无权图），visited 集合防重复。关键：入队时就标记 visited，不要出队时才标记。
from collections import deque, defaultdict

graph = defaultdict(list)
for u, v in edges:
    graph[u].append(v)
    graph[v].append(u)   # 无向图双向建边

def bfs(start, end):
    q = deque([(start, 0)])
    visited = {start}
    while q:
        node, dist = q.popleft()
        if node == end: return dist
        for nei in graph[node]:
            if nei not in visited:
                visited.add(nei)         # 入队即标记
                q.append((nei, dist + 1))
    return -1
Dijkstra（带权有向图，权重非负）
最小堆 + dist 数组。弹出时若距离已过期（d > dist[u]）直接跳过，保证每个节点只被松弛一次。
import heapq

def dijkstra(n, graph, src):
    # graph[u] = [(v, w), ...]
    dist = [float('inf')] * n
    dist[src] = 0
    heap = [(0, src)]       # (当前距离, 节点)
    while heap:
        d, u = heapq.heappop(heap)
        if d > dist[u]: continue    # 过期节点，跳过
        for v, w in graph[u]:
            if dist[u] + w < dist[v]:
                dist[v] = dist[u] + w
                heapq.heappush(heap, (dist[v], v))
    return dist
拓扑排序（Kahn 算法，BFS）
入度为 0 的节点先入队，每次出队一个节点并将其邻居入度 -1，新产生的入度 0 节点继续入队。最终若处理节点数 < n，说明有环。
from collections import deque, defaultdict

def topoSort(n, edges):
    graph = defaultdict(list)
    indegree = [0] * n
    for u, v in edges:
        graph[u].append(v)
        indegree[v] += 1

    q = deque(i for i in range(n) if indegree[i] == 0)
    order = []
    while q:
        u = q.popleft()
        order.append(u)
        for v in graph[u]:
            indegree[v] -= 1
            if indegree[v] == 0:
                q.append(v)
    return order if len(order) == n else []   # 长度不等 → 有环
并查集（路径压缩 + 按秩合并）
均摊 O(α(n)) ≈ O(1)。适合动态合并连通分量、判断是否成环、统计连通块数量。
class UnionFind:
    def __init__(self, n):
        self.parent = list(range(n))
        self.rank = [0] * n
        self.count = n              # 连通分量数

    def find(self, x):
        if self.parent[x] != x:
            self.parent[x] = self.find(self.parent[x])   # 路径压缩
        return self.parent[x]

    def union(self, x, y):
        px, py = self.find(x), self.find(y)
        if px == py: return False   # 已连通（加入此边会成环）
        if self.rank[px] < self.rank[py]: px, py = py, px
        self.parent[py] = px
        if self.rank[px] == self.rank[py]: self.rank[px] += 1
        self.count -= 1
        return True

    def connected(self, x, y):
        return self.find(x) == self.find(y)

# Top K 最大值：维护大小为 K 的最小堆
def topK(nums, k):
    heap = nums[:k]
    heapq.heapify(heap)
    for x in nums[k:]:
        if x > heap[0]:
            heapq.heapreplace(heap, x)   # 比堆顶大才替换
    return heap                           # 堆中即 Top K
双堆求实时中位数
大根堆存左半（较小值），小根堆存右半（较大值），保持 |lo| - |hi| <= 1。中位数 = 堆顶或两堆顶均值。
class MedianFinder:
    def __init__(self):
        self.lo = []    # 最大堆（存负值）：左半
        self.hi = []    # 最小堆：右半

    def addNum(self, num):
        heapq.heappush(self.lo, -num)                       # 先入左堆
        heapq.heappush(self.hi, -heapq.heappop(self.lo))    # 左堆最大值 → 右堆
        if len(self.lo) < len(self.hi):                     # 保持左堆 >= 右堆
            heapq.heappush(self.lo, -heapq.heappop(self.hi))

    def findMedian(self):
        if len(self.lo) > len(self.hi):
            return float(-self.lo[0])
        return (-self.lo[0] + self.hi[0]) / 2.0

Trie 树（前缀树）

字符串前缀查找 O(L) · 多词同时搜索
适用场景
单词搜索 / 自动补全 / 前缀统计
时间复杂度
插入/查找 O(L)，L = 字符串长度
核心字段
children 字典 + is_end 标记
每个节点代表一个字符，从根到某节点的路径表示一个前缀。is_end=True 表示该路径是完整单词。
Trie 完整实现
class TrieNode:
    def __init__(self):
        self.children = {}      # char → TrieNode
        self.is_end = False

class Trie:
    def __init__(self):
        self.root = TrieNode()

    def insert(self, word):
        node = self.root
        for c in word:
            if c not in node.children:
                node.children[c] = TrieNode()
            node = node.children[c]
        node.is_end = True

    def search(self, word):
        node = self.root
        for c in word:
            if c not in node.children: return False
            node = node.children[c]
        return node.is_end          # 必须是完整单词

    def startsWith(self, prefix):
        node = self.root
        for c in prefix:
            if c not in node.children: return False
            node = node.children[c]
        return True                 # 只要前缀存在即可
Trie + DFS — 单词搜索 II（board 中同时找多个词）
def findWords(board, words):
    trie = Trie()
    for w in words: trie.insert(w)

    m, n = len(board), len(board[0])
    res = set()

    def dfs(node, i, j, path):
        c = board[i][j]
        if c not in node.children: return
        nxt = node.children[c]
        path += c
        if nxt.is_end: res.add(path)
        board[i][j] = '#'       # 标记已访问
        for di, dj in [(-1,0),(1,0),(0,-1),(0,1)]:
            ni, nj = i + di, j + dj
            if 0 <= ni < m and 0 <= nj < n and board[ni][nj] != '#':
                dfs(nxt, ni, nj, path)
        board[i][j] = c         # 恢复

    for i in range(m):
        for j in range(n):
            dfs(trie.root, i, j, '')
    return list(res)

线段树 / 树状数组（BIT）

区间查询 + 动态更新，O(log n)
适用场景
区间和/最值 + 频繁单点/区间更新
vs 前缀和
前缀和不支持更新；线段树/BIT 支持
选择原则
只需前缀和+单点更新 → BIT（代码更短）
前缀和静态高效，但有更新就失效。有频繁单点更新用 BIT；需区间更新用线段树（加懒标记）。
树状数组 BIT（单点更新 + 前缀和查询）
class BIT:
    def __init__(self, n):
        self.n = n
        self.tree = [0] * (n + 1)   # 下标从 1 开始

    def update(self, i, delta):     # 单点更新，i 从 1 开始
        while i <= self.n:
            self.tree[i] += delta
            i += i & (-i)           # 跳到父节点（加上最低位）

    def query(self, i):             # 前缀和 [1..i]
        s = 0
        while i > 0:
            s += self.tree[i]
            i -= i & (-i)           # 跳到前一区段（去掉最低位）
        return s

    def rangeQuery(self, l, r):     # 区间和 [l..r]
        return self.query(r) - self.query(l - 1)
线段树（单点更新 + 区间和查询）
class SegTree:
    def __init__(self, nums):
        self.n = len(nums)
        self.tree = [0] * (4 * self.n)
        self._build(nums, 0, 0, self.n - 1)

    def _build(self, nums, node, l, r):
        if l == r:
            self.tree[node] = nums[l]; return
        mid = (l + r) // 2
        self._build(nums, 2*node+1, l, mid)
        self._build(nums, 2*node+2, mid+1, r)
        self.tree[node] = self.tree[2*node+1] + self.tree[2*node+2]

    def update(self, idx, val, node=0, l=0, r=None):
        if r is None: r = self.n - 1
        if l == r:
            self.tree[node] = val; return
        mid = (l + r) // 2
        if idx <= mid: self.update(idx, val, 2*node+1, l, mid)
        else:          self.update(idx, val, 2*node+2, mid+1, r)
        self.tree[node] = self.tree[2*node+1] + self.tree[2*node+2]

    def query(self, ql, qr, node=0, l=0, r=None):
        if r is None: r = self.n - 1
        if ql <= l and r <= qr: return self.tree[node]  # 完全覆盖
        if qr < l or r < ql:   return 0                 # 不相交
        mid = (l + r) // 2
        return (self.query(ql, qr, 2*node+1, l, mid) +
                self.query(ql, qr, 2*node+2, mid+1, r))

def singleNumber(nums):
    return reduce(lambda a, b: a ^ b, nums)

# 两个数各出现一次，其余出现两次（LC 260）
def singleNumberIII(nums):
    xor = reduce(lambda a, b: a ^ b, nums)   # xor = a ^ b
    diff = xor & (-xor)          # 找 a 与 b 不同的最低位
    a = 0
    for x in nums:
        if x & diff: a ^= x      # 按该位分组，只对一组 XOR
    return [a, xor ^ a]
状态压缩 DP — 访问所有节点最短路（TSP 骨架）
用整数二进制位表示"已访问集合"，将指数级枚举变为可记忆化状态。状压 DP 核心是：mask 代表集合，遍历时枚举下一步加入哪个节点。
n = 4   # 节点数
INF = float('inf')
dp = [[INF] * n for _ in range(1 << n)]
dp[1][0] = 0    # 从节点 0 出发，初始只访问了 0（bit 0 = 1）

for mask in range(1 << n):
    for u in range(n):
        if not (mask >> u & 1): continue    # u 不在当前集合中
        if dp[mask][u] == INF: continue
        for v in range(n):
            if mask >> v & 1: continue      # v 已访问过
            new_mask = mask | (1 << v)
            dp[new_mask][v] = min(dp[new_mask][v],
                                  dp[mask][u] + dist[u][v])

# 答案：访问所有节点后回到起点
ans = min(dp[(1 << n) - 1][i] + dist[i][0] for i in range(1, n))
枚举 mask 的所有子集
sub = mask
while sub:
    process(sub)
    sub = (sub - 1) & mask    # 获取下一个严格子集
经典题
易只出现一次的数字（LC 136）
全员 XOR 即得答案。a^a=0，a^0=a，XOR 满足交换律结合律，成对消除。
中数字范围按位与（LC 201）
找 m 和 n 的公共前缀：同时右移直到 m==n，记录位移次数，结果再左移复原。
中汉明距离总和（LC 477）
按位统计：第 k 位有 ones 个 1，该位贡献 ones*(n-ones)。32 位逐位处理，O(32n)。
难最短超级串（LC 943）
状压 DP：dp[mask][i] = 访问 mask 中单词、最后一个是 i 时的最短拼接长度。预处理重叠长度。
复习建议
第二梯队（①~④）高频必掌握：单调栈的"弹出时机"、回溯的"去重条件"（used[i-1]==False）、Dijkstra 的"跳过过期节点"、双堆求中位数的"平衡逻辑"。
第三梯队（⑤~⑦）加分掌握：Trie 和 XOR 技巧考频最高，优先拿下；BIT 代码短优先于线段树；状压 DP 理解 mask 含义和子集枚举即可。
