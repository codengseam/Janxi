滑动窗口（最长/最短子串）
右扩左缩维持窗口合法性。关键：什么时候移 left，ans 在哪里更新
left = 0
window = {}
for right in range(len(s)):
    # 1. 右边界入窗口
    window[s[right]] = window.get(s[right], 0) + 1
    # 2. 窗口不合法 → 收缩左边界
    while len(window) > k:
        window[s[left]] -= 1
        if window[s[left]] == 0: del window[s[left]]

def twoSum(nums, target):
    seen = {}           # 值 → 下标
    for i, x in enumerate(nums):
        need = target - x
        if need in seen:
            return [seen[need], i]
        seen[x] = i     # 先查后存，避免自配自己

from collections import defaultdict, Counter

def groupAnagrams(strs):
    groups = defaultdict(list)
    for s in strs:
        key = tuple(sorted(s))   # 或用 Counter 的 items
        groups[key].append(s)
    return list(groups.values())
前缀和 + 哈希（组合拳）
def subarraySum(nums, k):
    count, prefix = 0, 0
    freq = {0: 1}       # 前缀和 → 出现次数
    for x in nums:
        prefix += x
        count += freq.get(prefix - k, 0)
        freq[prefix] = freq.get(prefix, 0) + 1
    return count

反转链表（迭代）— 必须背熟
prev=None 做新尾，每步：存 next → 断开反向 → 整体前移。所有反转变体的基础。
def reverse(head):
    prev, curr = None, head
    while curr:
        nxt = curr.next   # ① 保存
        curr.next = prev  # ② 反向
        prev = curr       # ③ 前移
        curr = nxt
    return prev           # 新头

def mergeTwoLists(l1, l2):
    dummy = cur = ListNode(0)
    while l1 and l2:
        if l1.val <= l2.val: cur.next = l1; l1 = l1.next
        else:                cur.next = l2; l2 = l2.next
        cur = cur.next
    cur.next = l1 or l2
    return dummy.next

def traverse(root):
    if not root: return

    # 【前序位置】root 已知，子树未知 → 适合传递信息给子节点
    preorder_logic(root)

    traverse(root.left)

    # 【中序位置】左子树处理完 → BST 场景常用
    inorder_logic(root)

    traverse(root.right)

    # 【后序位置】两棵子树都处理完 → 适合汇聚子树信息
    postorder_logic(root)

from collections import deque

def levelOrder(root):
    if not root: return []
    q, res = deque([root]), []
    while q:
        level = []
        for _ in range(len(q)):     # 固定本层大小！
            node = q.popleft()
            level.append(node.val)
            if node.left:  q.append(node.left)
            if node.right: q.append(node.right)
        res.append(level)
    return res

def preorderIter(root):
    if not root: return []
    stack, res = [root], []
    while stack:
        node = stack.pop()
        res.append(node.val)
        if node.right: stack.append(node.right)  # 右先入
        if node.left:  stack.append(node.left)   # 左后入（后出先访问）
    return res

编辑距离
def minDistance(w1, w2):
    m, n = len(w1), len(w2)
    dp = [[0]*(n+1) for _ in range(m+1)]
    for i in range(m+1): dp[i][0] = i
    for j in range(n+1): dp[0][j] = j
    for i in range(1, m+1):
        for j in range(1, n+1):
            if w1[i-1] == w2[j-1]:
                dp[i][j] = dp[i-1][j-1]
            else:
                dp[i][j] = min(dp[i-1][j], dp[i][j-1], dp[i-1][j-1]) + 1
    return dp[m][n]

二分查找

有序 / 单调 / 答案可二分 → O(log n)
适用条件
有序 / 单调 / 答案满足单调性
时间复杂度
O(log n)，每次排除一半
易错点
边界 ±1 · while ≤ 还是 < · mid 溢出
标准模板（左闭右闭）
统一用左闭右闭。while 用 ≤；mid 用 left+(right-left)//2 防溢出。
def binary_search(nums, target):
    left, right = 0, len(nums) - 1
    while left <= right:
        mid = left + (right - left) // 2   # 防溢出
        if nums[mid] == target:   return mid
        elif nums[mid] < target:  left = mid + 1
        else:                     right = mid - 1
    return -1
找左边界（第一个 ≥ target 的位置）
def lower_bound(nums, target):
    left, right = 0, len(nums) - 1
    while left <= right:
        mid = left + (right - left) // 2
        if nums[mid] < target: left = mid + 1
        else:                  right = mid - 1    # >= target 都往左收
    return left   # 插入点，即第一个 >= target 的下标
找右边界（最后一个 ≤ target 的位置）
def upper_bound(nums, target):
    left, right = 0, len(nums) - 1
    while left <= right:
        mid = left + (right - left) // 2
        if nums[mid] <= target: left = mid + 1    # <= target 都往右扩
        else:                   right = mid - 1
    return right  # 最后一个 <= target 的下标
答案二分（二分答案本身）
当答案满足单调性（越大越容易满足/越小越容易满足），对答案值域做二分
left, right = 下界, 上界
